```python
#include <iostream>
#include <string>
#include <memory.h>
using namespace std;
int solution(string s)
{
    int answer = 0;

    if (s.size() != 0)
        answer=  1;
    for (int i = 0; i <s.size(); i++)
    {
        string s_temp = "";
        s_temp += s[i];
      
        for(int j = i + 1; j < s.size(); j++)
        {
            
            s_temp += s[j];
            int n =s_temp.size();
            if (answer > n)
                continue;
            bool flag = true;
            for(int k = 0; k <  n / 2; k++)
            {
                if (s_temp[k] != s_temp[n - k - 1])
                {
                    flag = false;
                        break;
                }
            }
            if (flag)
                answer = max(answer, n);
        }
    }
    return answer;
}
```

모든 substring에 대하여 팰린드롬 검사를 한다.

이 문제는 시간복잡도가 O(n^3)이다.
